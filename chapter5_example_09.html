import matplotlib.pyplot as plt
import numpy as np
from scipy.stats import norm

n = 100
#    
p = 0.05
#   
mu = n * p
sigma = np.sqrt(n * p * (1-p))
k = 2

x = np.linspace(mu - 4*sigma, mu + 4*sigma, 100)
pdf = norm.pdf(x, mu, sigma)

plt.plot(x, pdf)
plt.fill_between(x[x>=k], pdf[x>=k])
plt.title(f"P({k}<=X) (n={n}, p={p})")
plt.xlabel("x")
plt.ylabel("P(X=x)")
plt.show()
