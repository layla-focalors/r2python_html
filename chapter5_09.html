#  8 / (5.29), p180
from scipy.stats import expon

mu = 3
x = 9
x1 = 6

prob = expon.cdf(x, scale=mu)
print(f" 9   : {round((prob), 4)}")

prob = expon.cdf(x1, scale=mu) - expon.cdf(x, scale=mu)
print(f" 6 9   : {round((abs(prob)), 4)}")

 6 9   : 0.0855

#  8-1 / (5.29), p180 + 
import matplotlib.pyplot as plt
import numpy as np
from scipy.stats import expon

plt.rc('font', family='Malgun Gothic')

mu = 3
x = np.linspace(0, 15, 1000)
y = expon.pdf(x, scale=mu)

fig, ax = plt.subplots()
ax.plot(x, y)
ax.fill_between(x, y, where=(x <= 9), color='red', alpha=0.2)
ax.set_xlabel('  \nP(X <= 9)')
ax.set_ylabel('P(X)')
ax.set_title(' 9  ')

plt.show()

import matplotlib.pyplot as plt
